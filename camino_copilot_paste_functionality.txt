Camino Copilot Paste - Functionality Reference

Purpose
This file documents the current extension behavior so another project or agent can recreate it without reading or changing the original source files.

What the extension does
- Registers a command named testcopilot.start.
- When the command runs, it tries to open Copilot Chat in VS Code.
- If a file named test.png exists next to the extension entry file, it copies that image to the clipboard.
- After Copilot Chat is open, it tries to paste the image into the chat input.
- It then types a prompt: "Analyze this issue from the image and fix it."
- If clipboard or paste actions fail, it shows a message so the user can paste manually.
- Logging is written to an output channel named Camino Paste.

Important manifest requirements
- main should point to ./extension.js.
- activationEvents should include onCommand:testcopilot.start.
- contributes.commands should register testcopilot.start with a user-facing title.

Implementation flow
1. Activation
- On activation, the extension creates an output channel for logs.
- It registers the testcopilot.start command.

2. Command execution
- The command captures the start time.
- It checks for test.png in the same folder as extension.js.
- If the image exists, it starts copying the file to the clipboard.
- In parallel, it opens Copilot Chat and waits briefly for the UI to be ready.
- If the clipboard copy succeeds, it attempts paste using editor.action.clipboardPasteAction first.
- If that fails, it tries the paste command.
- If both paste attempts fail, it informs the user to paste manually.
- Finally, it types the prompt into the active editor/chat input and submits it.

3. Clipboard behavior by platform
- Windows: uses PowerShell with System.Drawing and System.Windows.Forms to place the image on the clipboard.
- macOS: uses osascript to read the file as a TIFF picture and set the clipboard.
- Linux: tries wl-copy first, then xclip.
- If no supported clipboard method exists, it returns an error.

4. Copilot Chat opening behavior
- The code tries github.copilot-chat.focus first.
- If that fails, it tries workbench.action.chat.open.
- If neither works, the function returns false.

5. Error handling
- Clipboard copy failures do not stop the whole command.
- If the image cannot be copied, the user is warned but the prompt typing still continues.
- If automatic prompt typing fails, the user is informed to paste or type the prompt manually.
- The output channel collects debug messages for troubleshooting.

const fs = require('fs');
const path = require('path');
const { execFile } = require('child_process');
const vscode = require('vscode');
let outputChannel = null;

function logDebug(msg) {
  const ts = new Date().toISOString();
  const line = `[${ts}] ${msg}`;
  if (outputChannel) outputChannel.appendLine(line);
  else console.log(line);
}

function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

async function openCopilotChat() {
  const commandsToTry = [
    'github.copilot-chat.focus',
    'workbench.action.chat.open'
  ];

  for (const commandId of commandsToTry) {
    try {
      await vscode.commands.executeCommand(commandId);
      return true;
    } catch {
      // Try the next command.
    }
  }

  return false;
}

function copyFileToClipboard(filePath) {
  const platform = process.platform;

  if (platform === 'win32') {
    const escapedPath = filePath.replace(/'/g, "''");
    const script = [
      'Add-Type -AssemblyName System.Drawing',
      'Add-Type -AssemblyName System.Windows.Forms',
      `$img = [System.Drawing.Image]::FromFile('${escapedPath}')`,
      '[System.Windows.Forms.Clipboard]::SetImage($img)'
    ].join('; ');

    return new Promise((resolve, reject) => {
      execFile(
        'powershell.exe',
        ['-NoProfile', '-STA', '-Command', script],
        (error, stdout, stderr) => {
          if (error) {
            reject(new Error(error.message + '\n' + stderr));
            return;
          }

          resolve();
        }
      );
    });
  }

  if (platform === 'darwin') {
    // Use AppleScript/osascript to read the file and set it as a TIFF picture on clipboard.
    const script = `set the clipboard to (read (POSIX file "${filePath}") as TIFF picture)`;

    return new Promise((resolve, reject) => {
      execFile('osascript', ['-e', script], (error, stdout, stderr) => {
        if (error) {
          reject(new Error(error.message + '\n' + stderr));
          return;
        }

        resolve();
      });
    });
  }

  // Linux (try wl-copy then xclip)
  if (platform === 'linux') {
    return new Promise((resolve, reject) => {
      // check for wl-copy
      execFile('which', ['wl-copy'], (errWl) => {
        if (!errWl) {
          // use wl-copy
          const cmd = `wl-copy --type image/png < '${filePath.replace(/'/g, "'\\''")}'`;
          execFile('bash', ['-lc', cmd], (err, stdout, stderr) => {
            if (err) {
              reject(new Error(err.message + '\n' + stderr));
              return;
            }

            resolve();
          });
          return;
        }

        // check for xclip
        execFile('which', ['xclip'], (errX) => {
          if (!errX) {
            const cmd = `xclip -selection clipboard -t image/png -i '${filePath.replace(/'/g, "'\\''")}'`;
            execFile('bash', ['-lc', cmd], (err2, stdout2, stderr2) => {
              if (err2) {
                reject(new Error(err2.message + '\n' + stderr2));
                return;
              }

              resolve();
            });
            return;
          }

          reject(new Error('No clipboard image tool found: install wl-clipboard (wl-copy) or xclip'));
        });
      });
    });
  }

  return Promise.reject(new Error('Unsupported platform for image clipboard'));
}

function activate(context) {
  outputChannel = vscode.window.createOutputChannel('Camino Paste');
  logDebug('activate');
  const disposable = vscode.commands.registerCommand('testcopilot.start', async () => {
    const start = Date.now();
    logDebug('command invoked');

    // We rely on copying the image to the clipboard and using VS Code's
    // `type` command to send the prompt. No native automation library is
    // required here to avoid native build issues.

    const imagePath = path.join(__dirname, 'test.png');

    logDebug('checking image path exists: ' + imagePath);
    const copyPromise = (fs.existsSync(imagePath))
      ? (async () => {
          logDebug('starting copyFileToClipboard');
          const r = await copyFileToClipboard(imagePath).catch(err => {
            logDebug('copyFileToClipboard failed: ' + err.message);
            vscode.window.showWarningMessage(`Could not copy test.png to clipboard: ${err.message}`);
            return null;
          });
          logDebug('copyFileToClipboard finished');
          return r;
        })()
      : null;

    logDebug('calling openCopilotChat');
    const openPromise = openCopilotChat();
    await Promise.all([openPromise, sleep(300)]);
    logDebug('openCopilotChat completed');

    if (copyPromise) {
      logDebug('waiting for copyPromise');
      await copyPromise;
      logDebug('copyPromise resolved');
      await sleep(100);

      let pasted = false;
      logDebug('attempting editor.action.clipboardPasteAction');
      try {
        await vscode.commands.executeCommand('editor.action.clipboardPasteAction');
        pasted = true;
        logDebug('editor.action.clipboardPasteAction succeeded');
      } catch (e) {
        logDebug('editor.action.clipboardPasteAction failed');
      }

      if (!pasted) {
        logDebug('attempting paste command');
        try {
          await vscode.commands.executeCommand('paste');
          pasted = true;
          logDebug('paste command succeeded');
        } catch (e) {
          logDebug('paste command failed');
        }
      }

      if (!pasted) {
        logDebug('no paste succeeded; prompting user');
        vscode.window.showInformationMessage('Image copied to clipboard; please paste manually into Copilot Chat (Ctrl+V).');
      }
    }

    const prompt = 'Analyze this issue from the image and fix it.';

    try {
      logDebug('typing prompt');
      await vscode.commands.executeCommand('type', { text: prompt + '\n' });
      logDebug('typing prompt finished, total: ' + (Date.now() - start) + 'ms');
    } catch (err) {
      logDebug('typing prompt failed: ' + (err && err.message));
      vscode.window.showInformationMessage('Could not auto-type prompt; please paste or type: ' + prompt);
    }
  });

  context.subscriptions.push(disposable);
}

function deactivate() {}

module.exports = {
  activate,
  deactivate
};

Recreation notes for another project
- Copy the logic into a new extension entry file if needed.
- Keep the command id testcopilot.start consistent across package.json and the code.
- If you want the image feature, place test.png beside the extension entry file.
- If you only want the text prompt behavior, you can keep the same openCopilotChat and type flow and remove the clipboard section.
- The code is designed to be self-contained and not depend on a native automation library.
